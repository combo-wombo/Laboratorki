-- Создание таблицы "Законодатели"
CREATE TABLE Законодатели (
    Идентификатор INT PRIMARY KEY,
    Имя VARCHAR(255),
    Адрес VARCHAR(255),
    Домашний_телефон VARCHAR(15),
    Служебный_телефон VARCHAR(15)
);

-- Создание таблицы "Комиссии"
CREATE TABLE Комиссии (
    Идентификатор INT PRIMARY KEY,
    Название VARCHAR(255),
    Профиль VARCHAR(255),
    Идентификатор_председателя INT,
    FOREIGN KEY (Идентификатор_председателя) REFERENCES Законодатели(Идентификатор)
);

-- Создание таблицы "Члены_Комиссий" для связи многие-ко-многим между Законодатели и Комиссии
CREATE TABLE Члены_Комиссий (
    Идентификатор INT PRIMARY KEY,
    Идентификатор_законодателя INT,
    Идентификатор_комиссии INT,
    FOREIGN KEY (Идентификатор_законодателя) REFERENCES Законодатели(Идентификатор),
    FOREIGN KEY (Идентификатор_комиссии) REFERENCES Комиссии(Идентификатор)
);

-- Создание таблицы "Заседания"
CREATE TABLE Заседания (
    Идентификатор INT PRIMARY KEY,
    Дата_и_время DATETIME,
    Идентификатор_комиссии INT,
    FOREIGN KEY (Идентификатор_комиссии) REFERENCES Комиссии(Идентификатор)
);
